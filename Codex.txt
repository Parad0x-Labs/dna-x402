/// The Verifying Key generated by your Circom circuit.

pub mod verifying_key;
use verifying_key::VERIFYING_KEY;

/// Seed used to derive Nullifier PDAs.
/// Prevents double-spending by tracking consumed notes.
const NULLIFIER_SEED: &[u8] = b"pdx_nullifier";

/// Seed for the Protocol Vault PDA.
/// This account MUST be pre-funded to pay relayers.
const VAULT_SEED: &[u8] = b"pdx_vault";

/// Fee paid to the Relayer for processing the transaction.
/// 0.05 SOL (in lamports).
const RELAYER_FEE: u64 = 50_000_000;

// -----------------------------------------------------------------------------
// 📦 INSTRUCTION DESERIALIZATION
// -----------------------------------------------------------------------------

#[derive(BorshSerialize, BorshDeserialize, Debug)]
pub enum DarkInstruction {
    /// Standard Deposit (Public)
    /// Users deposit SOL/Tokens to mint a private note.
    Deposit {
        amount: u64,
        commitment: [u8; 32]
    },

    /// Private Transfer (The "Null" Transaction)
    /// ZK-Verified, Relayer-Paid, Encrypted Payload.
    Transfer {
        proof: Vec<u8>,             // Groth16 Proof (A[64], B[128], C[64])
        root: [u8; 32],             // Merkle Root
        nullifier_asset: [u8; 32],  // Unique ID for Asset Note
        nullifier_fee: [u8; 32],    // Unique ID for Fee Note
        new_commitment: [u8; 32],   // New Note Output
        asset_id_hash: [u8; 32],    // INTEGRITY LOCK: keccak256(nebula_payload)
        nebula_payload: Vec<u8>     // Encrypted Data (for recipient)
    },
}

// -----------------------------------------------------------------------------
// 🚀 ENTRYPOINT
// -----------------------------------------------------------------------------

entrypoint!(process_instruction);

pub fn process_instruction(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    ix_data: &[u8],
) -> ProgramResult {
    // Attempt to parse instruction
    let ix = DarkInstruction::try_from_slice(ix_data)
        .map_err(|_| ProgramError::InvalidInstructionData)?;

    match ix {
        DarkInstruction::Deposit { amount, commitment } => {
            msg!("PDX_INFO: Deposit Request: {} lamports", amount);
            msg!("PDX_INFO: Commitment: {:?}", commitment);

            // In a real implementation, you would transfer funds from User -> Vault here.
            Ok(())
        }

        DarkInstruction::Transfer {
            proof,
            root,
            nullifier_asset,
            nullifier_fee,
            new_commitment,
            asset_id_hash,
            nebula_payload,
        } => process_transfer(
            program_id,
            accounts,
            proof,
            root,
            nullifier_asset,
            nullifier_fee,
            new_commitment,
            asset_id_hash,
            nebula_payload,
        ),
    }
}

// -----------------------------------------------------------------------------
// 🛡️ CORE LOGIC: PROCESS TRANSFER
// -----------------------------------------------------------------------------

fn process_transfer(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    proof: Vec<u8>,
    root: [u8; 32],
    nullifier_asset: [u8; 32],
    nullifier_fee: [u8; 32],
    new_commitment: [u8; 32],
    asset_id_hash: [u8; 32],
    nebula_payload: Vec<u8>,
) -> ProgramResult {
    let acc_iter = &mut accounts.iter();

    // --- 1. ACCOUNT DECODING ---

    let payer = next_account_info(acc_iter)?;          // [0] Relayer (Signer, receives fee)
    let null_asset_pda = next_account_info(acc_iter)?; // [1] Asset Nullifier PDA (Writable)
    let null_fee_pda = next_account_info(acc_iter)?;   // [2] Fee Nullifier PDA (Writable)
    let vault_pda = next_account_info(acc_iter)?;      // [3] Vault PDA (Source of fee)
    let system = next_account_info(acc_iter)?;         // [4] System Program

    // --- 2. HARDENING: VAULT HEALTH CHECK ---

    // Fail FAST if the relayer cannot be paid.
    // This protects the user from burning a proof for no reason.

    let (vault_addr, vault_bump) = Pubkey::find_program_address(&[VAULT_SEED], program_id);
    if *vault_pda.key != vault_addr {
        msg!("PDX_ERROR: Invalid Vault PDA provided.");
        return Err(ProgramError::InvalidArgument);
    }

    if **vault_pda.lamports.borrow() < RELAYER_FEE {
        msg!("PDX_ERROR: Vault is empty! Cannot pay Relayer Fee.");
        return Err(ProgramError::InsufficientFunds);
    }

    // --- 3. SECURITY: PAYLOAD INTEGRITY (Anti-Griefing) ---

    // The ZK proof verifies 'asset_id_hash'. We must ensure 'nebula_payload' matches it.
    let calculated_hash = keccak::hash(&nebula_payload);
    if calculated_hash.to_bytes() != asset_id_hash {
        msg!("PDX_CRITICAL: Payload Integrity Mismatch!");
        msg!("Expected: {:?}", asset_id_hash);
        msg!("Actual:   {:?}", calculated_hash.to_bytes());
        return Err(ProgramError::InvalidInstructionData);
    }

    // --- 4. SECURITY: INPUT VALIDATION ---

    validate_public_inputs(&root, &nullifier_asset, &nullifier_fee, &new_commitment, &asset_id_hash)?;

    // --- 5. SECURITY: ZK PROOF VERIFICATION (The Shield) ---

    // This MUST happen before any state change.
    let public_inputs = [root, nullifier_asset, nullifier_fee, new_commitment, asset_id_hash];
    if !verify_proof(&proof, &public_inputs)? {
        msg!("PDX_CRITICAL: ZK Proof Verification Failed.");
        return Err(ProgramError::InvalidArgument);
    }

    // --- 6. SECURITY: NULLIFIER UNIQUENESS ---

    // Ensure these notes haven't been spent before.
    verify_nullifier_unused(program_id, null_asset_pda, &nullifier_asset)?;
    verify_nullifier_unused(program_id, null_fee_pda, &nullifier_fee)?;

    // =====================================================================
    // ⚡ POINT OF NO RETURN: STATE MUTATION STARTS HERE
    // =====================================================================

    // --- 7. CONSUME NULLIFIERS (Burn the Notes) ---

    // We create the PDA accounts and mark them as "Consumed".
    consume_nullifier(program_id, payer, null_asset_pda, system, &nullifier_asset)?;
    consume_nullifier(program_id, payer, null_fee_pda, system, &nullifier_fee)?;

    // --- 8. PAY THE RELAYER (The Incentive) ---

    // Atomic transfer from Vault -> Relayer.
    msg!("PDX_INFO: Paying Relayer: {} lamports", RELAYER_FEE);
    invoke_signed(
        &system_instruction::transfer(
            vault_pda.key,
            payer.key,
            RELAYER_FEE
        ),
        &[vault_pda.clone(), payer.clone(), system.clone()],
        &[&[VAULT_SEED, &[vault_bump]]]
    )?;

    msg!("PDX_SUCCESS: Null Transaction Verified & Settled.");
    Ok(())
}

// -----------------------------------------------------------------------------
// 🛠️ HELPER FUNCTIONS
// -----------------------------------------------------------------------------

fn validate_public_inputs(
    root: &[u8; 32],
    nullifier_asset: &[u8; 32],
    nullifier_fee: &[u8; 32],
    new_commitment: &[u8; 32],
    asset_id_hash: &[u8; 32],
) -> ProgramResult {
    let inputs = [root, nullifier_asset, nullifier_fee, new_commitment, asset_id_hash];
    for (i, input) in inputs.iter().enumerate() {
        // Reject all-zero inputs (common attack vector)
        if input.iter().all(|&x| x == 0) {
            msg!("PDX_ERROR: Zero input detected at index {}", i);
            return Err(ProgramError::InvalidArgument);
        }
    }

    // Asset and Fee nullifiers must be distinct
    if nullifier_asset == nullifier_fee {
        msg!("PDX_ERROR: Identical Nullifiers.");
        return Err(ProgramError::InvalidArgument);
    }

    Ok(())
}

fn verify_nullifier_unused(
    program_id: &Pubkey,
    pda_account: &AccountInfo,
    nullifier: &[u8; 32],
) -> ProgramResult {
    let (expected_pda, _bump) = Pubkey::find_program_address(&[NULLIFIER_SEED, nullifier], program_id);

    if expected_pda != *pda_account.key {
        msg!("PDX_ERROR: Invalid Nullifier PDA address.");
        return Err(ProgramError::InvalidArgument);
    }

    // If account has data, it means it was already initialized (Spent)
    if !pda_account.data_is_empty() {
        msg!("PDX_ERROR: Double Spend Detected! Nullifier already used.");
        return Err(ProgramError::AccountAlreadyInitialized);
    }
    Ok(())
}

fn consume_nullifier(
    prog_id: &Pubkey,
    payer: &AccountInfo,
    pda: &AccountInfo,
    sys: &AccountInfo,
    null: &[u8; 32]
) -> ProgramResult {
    let (key, bump) = Pubkey::find_program_address(&[NULLIFIER_SEED, null], prog_id);

    if key != *pda.key {
        return Err(ProgramError::InvalidArgument);
    }
    // Create the account with exactly 1 byte of space.
    // This prevents "Fake Account" attacks where data is empty but account exists.
    invoke_signed(
        &system_instruction::create_account(
            payer.key,
            pda.key,
            Rent::default().minimum_balance(1), // Rent for 1 byte
            1,                                  // Space: 1 byte
            prog_id
        ),
        &[payer.clone(), pda.clone(), sys.clone()],
        &[&[NULLIFIER_SEED, null, &[bump]]]
    )?;

    // Explicitly write "1" to mark as consumed.
    let mut data = pda.try_borrow_mut_data()?;
    data[0] = 1;
    Ok(())
}

fn verify_proof(proof_bytes: &[u8], public_inputs: &[[u8; 32]]) -> Result<bool, ProgramError> {
    if proof_bytes.len() != 256 {
        msg!("PDX_ERROR: Invalid Proof Length.");
        return Err(ProgramError::InvalidArgument);
    }

    // 1. Parse Proof Points (A, B, C)
    let mut proof_a = [0u8; 64]; proof_a.copy_from_slice(&proof_bytes[0..64]);
    let a = G1::deserialize_compressed(&proof_a[..]).map_err(|_| ProgramError::InvalidArgument)?;

    let mut proof_b = [0u8; 128]; proof_b.copy_from_slice(&proof_bytes[64..192]);
    let b = ark_bn254::g2::G2Affine::deserialize_compressed(&proof_b[..]).map_err(|_| ProgramError::InvalidArgument)?;

    let mut proof_c = [0u8; 64]; proof_c.copy_from_slice(&proof_bytes[192..256]);
    let c = G1::deserialize_compressed(&proof_c[..]).map_err(|_| ProgramError::InvalidArgument)?;

    // 2. Prepare Public Inputs (Big Endian -> Little Endian conversion)
    let mut pub_inputs = Vec::new();
    for input in public_inputs {
        let mut bytes = [0u8; 32];
        bytes.copy_from_slice(input);
        // 🔧 ENDIANNESS FIX for SnarkJS Compatibility
        bytes.reverse();
        let field_element = ark_bn254::Fr::from_le_bytes_mod_order(&bytes);
        pub_inputs.push(field_element);
    }

    // 3. Verify
    match Groth16Verifier::verify(&pub_inputs, &VERIFYING_KEY, &a, &b, &c) {
        Ok(true) => Ok(true),
        _ => Ok(false),
    }
}
